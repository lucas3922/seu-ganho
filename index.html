import streamlit as st import pandas as pd from datetime import date # Configuração da Página st.set_page_config(page_title="Gestor Uber Pro", layout="wide") st.title("🚖 Gestor Financeiro: Uber + Casa") # --- SIDEBAR: GESTÃO DE CONTAS DA CASA --- st.sidebar.header("🏠 Contas da Casa (Mensal)") contas = { "Luz": st.sidebar.number_input("Luz", min_value=0.0, value=200.0), "Água": st.sidebar.number_input("Água", min_value=0.0, value=100.0), "Internet": st.sidebar.number_input("Internet", min_value=0.0, value=120.0), "Aluguel": st.sidebar.number_input("Aluguel", min_value=0.0, value=1200.0), "Mercado": st.sidebar.number_input("Mercado", min_value=0.0, value=800.0), "Outros": st.sidebar.number_input("Outros", min_value=0.0, value=0.0), } total_despesas = sum(contas.values()) dias_trabalho = st.sidebar.slider("Dias de trabalho no mês", 1, 31, 25) meta_diaria = total_despesas / dias_trabalho if dias_trabalho > 0 else 0 st.sidebar.divider() st.sidebar.metric("Total Despesas", f"R$ {total_despesas:,.2f}") st.sidebar.metric("Meta Diária Sugerida", f"R$ {meta_diaria:,.2f}") # --- CORPO PRINCIPAL: LANÇAMENTO DIÁRIO --- col1, col2 = st.columns(2) with col1: st.subheader("📝 Lançamento do Dia") data_hoje = st.date_input("Data", date.today()) ganho_bruto = st.number_input("Ganho Bruto Total (App)", min_value=0.0) km_rodado = st.number_input("KM Total Rodado", min_value=0.1) horas = st.number_input("Horas Trabalhadas", min_value=0.1) custo_km = st.number_input("Custo por KM (Combustível/Manutenção)", value=0.60) # Cálculos Automáticos custo_total = km_rodado * custo_km ganho_liquido = ganho_bruto - custo_total # Métricas do Lançamento km_bruto = ganho_bruto / km_rodado km_liquido = ganho_liquido / km_rodado hora_bruta = ganho_bruto / horas hora_liquida = ganho_liquido / horas with col2: st.subheader("📊 Performance do Lançamento") st.write(f"*Ganho Líquido:* R$ {ganho_liquido:,.2f}") st.write(f"*KM Bruto:* R$ {km_bruto:,.2f} / *KM Líquido:* R$ {km_liquido:,.2f}") st.write(f"*Hora Bruta:* R$ {hora_bruta:,.2f} / *Hora Líquida:* R$ {hora_liquida:,.2f}") progresso = (ganho_liquido / total_despesas) if total_despesas > 0 else 0 st.progress(min(progresso, 1.0)) st.caption(f"Este ganho abateu {progresso*100:.1f}% das despesas fixas.") # --- ABATIMENTO E SALDO --- st.divider() st.subheader("📉 Saldo Devedor do Mês") saldo_restante = total_despesas - ganho_liquido # Exemplo simplificado para 1 lançamento st.warning(f"Faltam *R$ {max(0.0, saldo_restante):,.2f}* para cobrir as contas da casa este mês.") if st.button("Salvar Lançamento Diário"): st.success("Dados simulados e salvos com sucesso! (Para banco de dados real, conecte a um CSV ou SQL)")
